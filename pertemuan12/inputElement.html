input element

element input html sering digunakan dalam pembuatan form di html,element input bisa ditampilkan dengan berbagai cara tergantung dari atribut type

type descripsi
type = "text" menampilkan 1 baris text input(bisa menginputkan lebih dari 1 jenis tipe data)
type = "radio" menampilkan radio button(untuk memilih 1 dari banyak pilihan)
type = "submit" menampilkan tombol submit atau kirim(untuk mengirimkan data dalam form)
type = "checkbox" menampilkan checkbox(unutk memilih nol atau banyak pilihan sekaligus)
type = "button" menampilkan tombol dapat di klik
type = "number" hanya menampilkan inputan berupa angka
type = "password" menampilkan bintang dari inputan berupa pw
type = "textarea" menampilkan area inputan yang lebih besar dari type text
